#include <linux/kernel.h>

{
    printk( "Hello Linux System Call! \n" );

}

/* app_hellocall.c  */

#include <errno.h>
_syscall0(int, hellocall)

int main()
{
    int i;
    printf( "Before System Call \n");
    i = hellocall();
    printf( "After System Call \n");
}
